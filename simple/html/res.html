控制台

res 参数控制多种返回值

在 app.js 中配置 
$.ajax('/res/', {
    res: {
        'wait': {
            msg: 'wait...'
        },
        'finsh': {
            msg: 'finish...'
        }
    }
}).done({
    msg: "ok"
}).fail({
    msg: 'error'
})
在浏览器中测试(你可以打开控制台查看 log) 
$.get('/res/',{
    _: 1
}).done(function (data) {
    console.log(data)
    // {"msg": "ok"}
})

$.get('/res/',{
    _: 0
}).done(function (data) {
    console.log(data)
    // {"msg": "error"}
})

$.get('/res/',{
    _: 'wait'
}).done(function (data) {
    console.log(data)
    // {"msg": "wait..."}
})

$.get('/res/',{
    _: 'finsh'
}).done(function (data) {
    console.log(data)
    // {"msg": "finsh..."}
})


$.get('/res/',{
    _: 'not set'
})
// 因为没有设置过 res.not set  所以会返回 415 错误
// GET http://127.0.0.1:18080/res/?_=not+set 415 (Unsupported Media Type)
